eval "$(ssh-agent -s)"
ssh-add ~/.ssh/id_ed25519
ssh-keyscan github.com > ~/.ssh/known_hosts
echo "PermitRootLogin yes" > /etc/ssh/sshd_config.d/toc.conf
cat ~/.ssh/id_ed25519.pub' > ./first_init.sh
